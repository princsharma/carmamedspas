"use client";

import { motion, useReducedMotion } from "framer-motion";
import { type ReactNode } from "react";

type RevealProps = {
  children: ReactNode;
  className?: string;
  delay?: number;
  direction?: "up" | "down" | "left" | "right" | "none";
  duration?: number;
};

const offsets = {
  up: { y: 40, x: 0 },
  down: { y: -40, x: 0 },
  left: { x: 40, y: 0 },
  right: { x: -40, y: 0 },
  none: { x: 0, y: 0 },
};

const ease = [0.22, 1, 0.36, 1] as const;

export function Reveal({
  children,
  className,
  delay = 0,
  direction = "up",
  duration = 0.7,
}: RevealProps) {
  const reduceMotion = useReducedMotion();
  const offset = offsets[direction];

  if (reduceMotion) {
    return <div className={className}>{children}</div>;
  }

  return (
    <motion.div
      className={className}
      initial={{ opacity: 0, x: offset.x, y: offset.y }}
      whileInView={{ opacity: 1, x: 0, y: 0 }}
      viewport={{ once: true, margin: "-40px", amount: 0.15 }}
      transition={{
        duration,
        delay,
        ease,
      }}
    >
      {children}
    </motion.div>
  );
}

/** Large rounded section containers — slide up from below on scroll */
export function RevealBox({
  children,
  className,
  delay = 0,
}: {
  children: ReactNode;
  className?: string;
  delay?: number;
}) {
  const reduceMotion = useReducedMotion();

  if (reduceMotion) {
    return <div className={className}>{children}</div>;
  }

  return (
    <motion.div
      className={className}
      initial={{ opacity: 0, y: 80 }}
      whileInView={{ opacity: 1, y: 0 }}
      viewport={{ once: true, margin: "-60px", amount: 0.12 }}
      transition={{ duration: 0.85, delay, ease }}
    >
      {children}
    </motion.div>
  );
}

export function Stagger({
  children,
  className,
  stagger = 0.1,
}: {
  children: ReactNode;
  className?: string;
  stagger?: number;
}) {
  const reduceMotion = useReducedMotion();

  if (reduceMotion) {
    return <div className={className}>{children}</div>;
  }

  return (
    <motion.div
      className={className}
      initial="hidden"
      whileInView="visible"
      viewport={{ once: true, margin: "-40px", amount: 0.1 }}
      variants={{
        hidden: {},
        visible: { transition: { staggerChildren: stagger } },
      }}
    >
      {children}
    </motion.div>
  );
}

export function StaggerItem({
  children,
  className,
}: {
  children: ReactNode;
  className?: string;
}) {
  const reduceMotion = useReducedMotion();

  if (reduceMotion) {
    return <div className={className}>{children}</div>;
  }

  return (
    <motion.div
      className={className}
      variants={{
        hidden: { opacity: 0, y: 64 },
        visible: {
          opacity: 1,
          y: 0,
          transition: { duration: 0.75, ease },
        },
      }}
    >
      {children}
    </motion.div>
  );
}

"use client";

import { motion, useScroll, useSpring, useReducedMotion } from "framer-motion";

export function ScrollProgress() {
  const reduceMotion = useReducedMotion();
  const { scrollYProgress } = useScroll();
  const scaleX = useSpring(scrollYProgress, {
    stiffness: 120,
    damping: 30,
    restDelta: 0.001,
  });

  if (reduceMotion) return null;

  return (
    <motion.div
      className="scroll-progress"
      style={{ scaleX }}
      aria-hidden="true"
    />
  );
}

"use client";

import { motion, useScroll, useTransform, useReducedMotion } from "framer-motion";
import { useState, useEffect, useCallback } from "react";
import { useConsultBooking } from "@/context/ConsultBookingContext";
import { BookConsultButton } from "./BookConsultButton";
import { BrandLogo } from "./BrandLogo";

const links = [
  { href: "#why", label: "Why" },
  { href: "#how", label: "How it Works" },
  { href: "#medications", label: "Medications" },
  { href: "#pricing", label: "Pricing" },
  { href: "#faq", label: "FAQ" },
];

export function Navbar() {
  const [scrolled, setScrolled] = useState(false);
  const [menuOpen, setMenuOpen] = useState(false);
  const { scrollY } = useScroll();
  const reduceMotion = useReducedMotion();
  const { openBooking } = useConsultBooking();

  const navScale = useTransform(scrollY, [0, 120], [1, 0.96]);

  useEffect(() => {
    const onScroll = () => setScrolled(window.scrollY > 24);
    onScroll();
    window.addEventListener("scroll", onScroll, { passive: true });
    return () => window.removeEventListener("scroll", onScroll);
  }, []);

  useEffect(() => {
    document.body.style.overflow = menuOpen ? "hidden" : "";
    return () => {
      document.body.style.overflow = "";
    };
  }, [menuOpen]);

  useEffect(() => {
    if (!menuOpen) return;
    const onKeyDown = (e: KeyboardEvent) => {
      if (e.key === "Escape") setMenuOpen(false);
    };
    window.addEventListener("keydown", onKeyDown);
    return () => window.removeEventListener("keydown", onKeyDown);
  }, [menuOpen]);

  const closeMenu = useCallback(() => setMenuOpen(false), []);

  const handleNavClick = () => closeMenu();

  const handleBookFromMenu = () => {
    closeMenu();
    openBooking();
  };

  const navInner = (
    <>
      <a href="#top" className="nav-logo" aria-label="CARMA Med Spa home" onClick={handleNavClick}>
        <BrandLogo variant="nav" priority />
      </a>

      <nav className="nav-links" aria-label="Primary">
        {links.map((link) => (
          <a key={link.href} href={link.href}>
            {link.label}
          </a>
        ))}
      </nav>

      <div className="nav-actions">
        <BookConsultButton className="nav-cta">Book Consult</BookConsultButton>

        <button
          type="button"
          className={`nav-toggle${menuOpen ? " nav-toggle--open" : ""}`}
          aria-expanded={menuOpen}
          aria-controls="mobile-nav"
          aria-label={menuOpen ? "Close menu" : "Open menu"}
          onClick={() => setMenuOpen((open) => !open)}
        >
          <span aria-hidden="true" />
          <span aria-hidden="true" />
          <span aria-hidden="true" />
        </button>
      </div>

      <div
        id="mobile-nav"
        className={`nav-mobile${menuOpen ? " nav-mobile--open" : ""}`}
        aria-hidden={!menuOpen}
      >
        <nav className="nav-mobile-links" aria-label="Mobile">
          {links.map((link) => (
            <a key={link.href} href={link.href} onClick={handleNavClick}>
              {link.label}
            </a>
          ))}
        </nav>
        <button type="button" className="btn btn-primary btn-full" onClick={handleBookFromMenu}>
          Book Consult
        </button>
      </div>
    </>
  );

  if (reduceMotion) {
    return (
      <header className="nav-shell">
        <div className={`nav-float${scrolled ? " nav-float--scrolled" : ""}${menuOpen ? " nav-float--menu-open" : ""}`}>
          {navInner}
        </div>
      </header>
    );
  }

  return (
    <header className="nav-shell">
      <motion.div
        className={`nav-float${scrolled ? " nav-float--scrolled" : ""}${menuOpen ? " nav-float--menu-open" : ""}`}
        style={{ scale: menuOpen ? 1 : navScale }}
      >
        {navInner}
      </motion.div>
    </header>
  );
}

}

export function ConsultBookingModal() {
  const { isOpen, preferredPlan, closeBooking } = useConsultBooking();
  const reduceMotion = useReducedMotion();
  const titleId = useId();
  const [step, setStep] = useState(0);
  const [form, setForm] = useState<FormData>(initialForm);
  const [error, setError] = useState<string | null>(null);
  const [submitting, setSubmitting] = useState(false);
  const [referenceId, setReferenceId] = useState<string | null>(null);

  const reset = useCallback(() => {
    setStep(0);
    setForm(initialForm);
    setError(null);
    setSubmitting(false);
    setReferenceId(null);
  }, []);

  useEffect(() => {
    if (isOpen) {
      setForm((prev) => ({
        ...prev,
        medication: preferredPlan ?? "unsure",
      }));
      document.body.style.overflow = "hidden";
    } else {
      document.body.style.overflow = "";
      const timer = window.setTimeout(reset, 300);
      return () => window.clearTimeout(timer);
    }
    return () => {
      document.body.style.overflow = "";
    };
  }, [isOpen, preferredPlan, reset]);

  useEffect(() => {
    if (!isOpen) return;
    const onKeyDown = (e: KeyboardEvent) => {
      if (e.key === "Escape") closeBooking();
    };
    window.addEventListener("keydown", onKeyDown);
    return () => window.removeEventListener("keydown", onKeyDown);
  }, [isOpen, closeBooking]);

  const updateField = <K extends keyof FormData>(key: K, value: FormData[K]) => {
    setForm((prev) => ({ ...prev, [key]: value }));
    setError(null);
  };

  const validateStep = () => {
    if (step === 0) {
      if (!form.firstName.trim() || !form.lastName.trim()) {
        setError("Please enter your first and last name.");
        return false;
      }
      if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(form.email)) {
        setError("Please enter a valid email address.");
        return false;
      }
      if (form.phone.replace(/\D/g, "").length < 10) {
        setError("Please enter a valid phone number.");
        return false;
      }
    }
    if (step === 1) {
      if (!form.preferredDate) {
        setError("Please choose a preferred date.");
        return false;
      }
      if (!form.preferredTime) {
        setError("Please choose a preferred time.");
        return false;
      }
    }
    return true;
  };

  const handleNext = () => {
    if (!validateStep()) return;
    setStep((s) => Math.min(s + 1, 2));
  };

        setError(data.error ?? "Something went wrong. Please try again.");
        return;

      setReferenceId(data.referenceId);
      setStep(3);
    } catch {
      setError("Unable to submit right now. Please try again.");
    } finally {
      setSubmitting(false);
    }
  };

    semaglutide: "Semaglutide",
    tirzepatide: "Tirzepatide",
    unsure: "Not sure yet",
  }[form.medication];

  return (
    <AnimatePresence>
      {isOpen && (
        <motion.div
          className="booking-overlay"
          initial={reduceMotion ? false : { opacity: 0 }}
          animate={{ opacity: 1 }}
          exit={{ opacity: 0 }}
          transition={{ duration: 0.25 }}
          onClick={closeBooking}
        >
          <motion.div
            role="dialog"
            aria-modal="true"
            aria-labelledby={titleId}
            className="booking-modal"
            initial={reduceMotion ? false : { opacity: 0, y: 24, scale: 0.98 }}
            animate={{ opacity: 1, y: 0, scale: 1 }}
            exit={{ opacity: 0, y: 16, scale: 0.98 }}

          >
            <div className="booking-modal-header">
              <div>
                <BrandLogo variant="modal" className="booking-modal-logo" />
                <span className="kicker">Free Consultation</span>
                <h2 id={titleId}>
                  {step === 3 ? "You're all set" : "Book your consult"}
                </h2>
              </div>
              <button
                type="button"
                className="booking-close"
                onClick={closeBooking}
                aria-label="Close booking form"
              >
                ×
              </button>
            </div>

            {step < 3 && (
              <div className="booking-steps" aria-label="Booking progress">
                {steps.map((label, i) => (
                  <div
                    key={label}
                    className={`booking-step${i <= step ? " booking-step--active" : ""}${i < step ? " booking-step--done" : ""}`}
                  >
                    <span>{i + 1}</span>
                    <em>{label}</em>
                  </div>
                ))}
              </div>
            )}

            {error && (

            )}

      )}

  );
}

"use client";

import {
  createContext,
  useCallback,
  useContext,
  useMemo,
  useState,
  type ReactNode,
} from "react";

export type MedicationPlan = "semaglutide" | "tirzepatide" | "unsure";

type ConsultBookingContextValue = {
  isOpen: boolean;
  preferredPlan: MedicationPlan | null;
  openBooking: (plan?: MedicationPlan) => void;
  closeBooking: () => void;
};

const ConsultBookingContext = createContext<ConsultBookingContextValue | null>(
  null,
);

export function ConsultBookingProvider({ children }: { children: ReactNode }) {
  const [isOpen, setIsOpen] = useState(false);
  const [preferredPlan, setPreferredPlan] = useState<MedicationPlan | null>(
    null,
  );

  const openBooking = useCallback((plan?: MedicationPlan) => {
    setPreferredPlan(plan ?? null);
    setIsOpen(true);
  }, []);

  const closeBooking = useCallback(() => {
    setIsOpen(false);
    setPreferredPlan(null);
  }, []);

  const value = useMemo(
    () => ({ isOpen, preferredPlan, openBooking, closeBooking }),
    [isOpen, preferredPlan, openBooking, closeBooking],
  );

  return (
    <ConsultBookingContext.Provider value={value}>
      {children}
    </ConsultBookingContext.Provider>
  );
}

export function useConsultBooking() {
  const context = useContext(ConsultBookingContext);
  if (!context) {
    throw new Error(
      "useConsultBooking must be used within ConsultBookingProvider",
    );
  }
  return context;
}

import { NextRequest, NextResponse } from "next/server";

type ConsultPayload = {
  firstName: string;
  lastName: string;
  email: string;
  phone: string;
  medication: string;
  preferredDate: string;
  preferredTime: string;
  notes?: string;
};

function isValidEmail(email: string) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

function isValidPhone(phone: string) {
  const digits = phone.replace(/\D/g, "");
  return digits.length >= 10;
}

export async function POST(request: NextRequest) {
  let body: Partial<ConsultPayload>;

  try {
    body = await request.json();
  } catch {
    return NextResponse.json({ error: "Invalid request body." }, { status: 400 });
  }

  const {
    firstName,
    lastName,
    email,
    phone,
    medication,
    preferredDate,
    preferredTime,
    notes,
  } = body;

  if (!firstName?.trim() || !lastName?.trim()) {
    return NextResponse.json(
      { error: "First and last name are required." },
      { status: 400 },
    );
  }

  if (!email?.trim() || !isValidEmail(email)) {
    return NextResponse.json(
      { error: "A valid email address is required." },
      { status: 400 },
    );
  }

  if (!phone?.trim() || !isValidPhone(phone)) {
    return NextResponse.json(
      { error: "A valid phone number is required." },
      { status: 400 },
    );
  }

  if (!medication || !["semaglutide", "tirzepatide", "unsure"].includes(medication)) {
    return NextResponse.json(
      { error: "Please select a medication preference." },
      { status: 400 },
    );
  }

  if (!preferredDate?.trim() || !preferredTime?.trim()) {
    return NextResponse.json(
      { error: "Preferred date and time are required." },
      { status: 400 },
    );
  }

  const referenceId = `CM-${Date.now().toString(36).toUpperCase()}`;

  // Ready for CRM / email integration — log submission in development.
  if (process.env.NODE_ENV === "development") {
    console.info("[consult booking]", {
      referenceId,
      firstName: firstName.trim(),
      lastName: lastName.trim(),
      email: email.trim(),
      phone: phone.trim(),
      medication,
      preferredDate: preferredDate.trim(),
      preferredTime: preferredTime.trim(),
      notes: notes?.trim() || null,
    });
  }

  return NextResponse.json({
    success: true,
    referenceId,
    message:
      "Your consult request has been received. Our care team will confirm your appointment within one business day.",
  });
}
